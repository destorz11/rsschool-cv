function generateHashtag (str) {
  if (!str || !str.trim()) {
    return false; // Return false if input is empty or only whitespace.
  } else {
    let arr = str.split(' ').filter((elem)=>elem.length>0).map((elem)=>elem[0].toUpperCase() + elem.slice(1))
    if (arr.join('').length<140){
      return '#' + arr.join('')
    } else {
      return false
    }

  }
}
